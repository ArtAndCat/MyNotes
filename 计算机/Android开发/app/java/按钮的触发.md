```Java
public boolean onCreateOptionsMenu(Menu menu) {
        getMenuInflater().inflate(R.menu.main,menu);
        return true;//返回true表示可以显示
    }
    //菜单响应事件
    public boolean onOptionsItemSelected(MenuItem item) {
        switch (item.getItemId()){
            case R.id.i_1:
                Toast.makeText(MainActivity.this, "点1", Toast.LENGTH_LONG).show();
               break;
            case R.id.item_2:
                Toast.makeText(MainActivity.this, "点2", Toast.LENGTH_LONG).show();
               break;
            default:
        }
        return true ;
    }
```
